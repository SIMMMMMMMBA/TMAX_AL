### 1. 신범

```c++
#include<stdio.h>
int c[2002][2002];
int score[2002];
int tot[2002];
int count[2002];
int N,T,P;
int i, j;
int main(int argc, char** argv)
{
    setbuf(stdout, NULL);
   int test_case;
   int T;
    char b[2];
    scanf("%d",&T);
   for(test_case = 1; test_case <= T; ++test_case)
   {
        scanf("%d %d %d",&N,&T,&P);
        P--;
        for(i=0;i<2000;++i)
        {
            score[i]=0;
            tot[i]=0;
            count[i]=0;
        }
        for(i=0;i<N;++i)
            for(j=0;j<T;++j)
            {
                gets(b);
                c[i][j]=b[0]-'0';
                score[j]+=-c[i][j];
                count[i]+=c[i][j];
            }
        for(i=0;i<N;++i)
            for(j=0;j<T;++j)
                tot[i] += (c[i][j]==1) ? score[j] : 0;
        int cnt=0;
        for(i=0;i<N;++i)
            if(tot[i]>tot[P] || (tot[i]==tot[P] && count[i]>count[P]) || (tot[i]==tot[P] && count[i]==count[P] && i<P))
               cnt++;
        printf("#%d %d %d\n",test_case,tot[P],cnt+1);
   }
   return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
```

### 2. 답지

```C++
#include <iostream>
#include <string.h>
using namespace std;
 
int participants[2001][2001];
int problems[2001];
 
int main(int argc, char** argv) {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
 
    int TC;
    cin >> TC;
    for (int test_case = 1; test_case <= TC; ++test_case) {
        int N, T, P;
        cin >> N >> T >> P;
        memset(problems, 0, sizeof(problems));
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < T; ++j) {
                cin >> participants[i][j];
                if (!participants[i][j]) {
                    problems[j]++;
                }
            }
        }
         
        P--;
        int rank = 1;
        int pscore = 0, pnum = 0;
        for (int q = 0; q < T; ++q) {
            if (participants[P][q]) {
                pscore += problems[q];
                pnum++;
            }
        }
 
        for (int i = 0; i < N; ++i) {
            int score = 0, num = 0;
            for (int j = 0; j < T; ++j) {
                if (participants[i][j]) {
                    score += problems[j];
                    num++;
                }
            }
         
            if (pscore > score)
                continue;
            if (pscore == score) {
                if (num < pnum)
                    continue;
                if (num == pnum) {
                    if (i >= P)
                        continue;
                }
            }
            rank++;
        }
 
        cout << "#" << test_case << " "  << pscore  << " " << rank << "\n";
    }
 
    return 0;
}
```
