

### 1. 신범

```c++
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[10001];
int main()
{
	int T, test;
	int i;
	int n;
	scanf("%d", &T);
	for (test = 1; test <= T; ++test)
	{
		int tot = 0;
		scanf("%d", &n);
		for (i = 0; i < n; i++)
			scanf("%d", &a[i]);
		sort(a, a + n);
		for (i = 0; i < n; i++)
			tot += a[i] + 1;
		tot += a[n - 1];
		printf("#%d %d\n",test, tot);
	}
	return 0;
}ㄴ
```
